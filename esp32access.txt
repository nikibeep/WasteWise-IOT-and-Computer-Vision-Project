#include <WiFi.h>
#include <WiFiClient.h>

const char* ssid = "Wokwi-GUEST";
const char* password = "";
WiFiServer server(80);

void setup() {
  Serial.begin(115200);
  delay(10);
  // Connect to WiFi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Connecting to WiFi...");
  }
  Serial.println("Connected to WiFi");
  Serial.println(WiFi.localIP());
  server.begin();
}

void loop() {
  
  WiFiClient client = server.available();
  if (client) {
    Serial.println("New client connected");
    bool dataReceived = false; // Flag to indicate if data was received
    while (client.connected()) {
      // Read data from the client and process it here
      if (client.available() >= sizeof(int)) {
        int receivedData;
        client.read((uint8_t*)&receivedData, sizeof(int));
        Serial.print("Received data: ");
        Serial.println(receivedData);
        dataReceived = true; // Set the flag to true when data is received
        // You can now process the receivedData integer as needed
      }
    }
    client.stop();
    Serial.println("Client disconnected");
    
    if (dataReceived) {
      Serial.println("Data received successfully.");
    } else {
      Serial.println("No data received.");
    }
  }
}
